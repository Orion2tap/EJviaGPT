# 快捷键
#### Mac
```bash
# 显示桌面 (自定义)
command + D
# 显示 launchpad (自定义)
command + ~
# 聚焦搜索  
command + 空格
# 最小化
command + M
# 退出
command + Q
# 偏好设置
command + ,
# 显示隐藏文件夹
command+shift+.

# Finder
# 从桌面打开finder
command+shift+G # 按路径进入隐藏文件夹

```

#### Atom
```bash
# 多行光标
control + shift + ↑↓←→
# 显示/隐藏侧边栏
command + \
# 弹出终端
control + ~
# markdown预览
shift + control + M
# 自动import
option+回车
# 折叠所有/展开所有
option + command + shift + {/}
```

#### Idea
```bash
# 字体放大/缩小
command + 滚轮
# 全局查找
shift + command + F
# 搜索文件
shift + shift
# 重命名
shift + F6
# 格式化
option + command + L
# 折叠/展开
command + -/+
# 折叠所有/展开所有
shift + command + -/+
# 上下行互换
shift + command + ↑/↓
# 查找
command + F

# 替换
command + R
# 自动代码片 下拉选择
# 显示类之间的关系
option + command + U
# 显示类中的所有方法
command + F12
# 转到方法的定义
command + 左键单击方法名
# 查看函数调用情况
右键函数名->Find Usages
# 重启Web服务器  
command + R (console窗口)
# 光标跳转到行头/行尾
command+←/→
# 删除光标左侧单词
option + delete
# 删除整行
command + delete
# 光标多行同步操作
shift + option + 左键单击
```

#### Chrome
```bash
# 退出当前应用程序
command+Q
# 剪切
command + X
# 隐藏
command + H(还会用)    
# 最小化
command + M(暂时不用)
# 关闭标签页
command + W
# 切换标签页
command + 1~command + 9
# 新建标签页
command + T
# google开发者工具
option + command + I
```

#### IINA
```bash
# 加速1.1倍
option+command+]
```

# 触摸板
```bash
# 常规
# 显示桌面                          
四指分开
# 打开launchpad
四指捏合
# 打开mission control
三指上推
# 打开日程     
右边缘三指左划

# 浏览器
# 前进                                
双指右划
# 后退                                
双指左划

# 分屏
# 打开mission control    
三指上推
# A左B右或B左A右   
下面窗口拖到顶部
# 切换屏幕       
三指左划/右划
```
